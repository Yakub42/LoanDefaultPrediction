# Confusion matrix method
def plot_conf(data):
    confusion_matrix = metrics.confusion_matrix(test_y, data)
    cm_display = metrics.ConfusionMatrixDisplay(confusion_matrix = confusion_matrix, display_labels=[0, 1])
    cm_display.plot()
    plt.show()

    else: return 0

OVERSAMPLING AND UNDERSAMPLING TECHNIQUES¶

Since the logistic regression model has the best performance, let us see if we can improve the model performance by resampling the data.

In [28]:
from imblearn.over_sampling import RandomOverSampler
from imblearn.under_sampling import RandomUnderSampler

oversampler = RandomOverSampler(sampling_strategy='minority', random_state=2)
x_over, y_over = oversampler.fit_resample(train_x_norm, train_y)

lr_over_proba = pd.Series(lr_model.fit(x_over, y_over).predict_proba(test_x_norm)[:, 1])

scores = []
for a in np.arange(0.2, 1, 0.01):
    scores.append(metrics.f1_score(test_y, lr_over_proba.apply(tune_proba ,args = (a,))))
    
plt.plot(np.arange(0.2,1, 0.01), scores)
print(np.arange(0.2, 1, 0.01)[scores.index(max(scores))], max(scores))
